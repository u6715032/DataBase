CREATE TABLE users (
    user_id SERIAL PRIMARY KEY,
    username VARCHAR(50) UNIQUE NOT NULL,
    password VARCHAR(255) NOT NULL, -- Must be long to store hashes!
    role VARCHAR(20) NOT NULL CHECK (role IN ('admin', 'staff', 'user')),
    member_id INT -- Optional: Link to your members table if the role is 'user'
);
